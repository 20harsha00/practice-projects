'use client';

import React, { useState } from 'react';
import { useRouter } from 'next/navigation';
import styles from './ApprovalCategories.module.css';
import {
    MdFolder,
    MdDescription,
    MdShoppingCart,
    MdSwapHoriz,
    MdAttachMoney,
    MdReceiptLong,
    MdAirlineSeatReclineNormal,
} from 'react-icons/md';
import { FaWrench } from 'react-icons/fa';
import { FiChevronRight } from 'react-icons/fi';

interface ApprovalCategory {
    id: string;
    name: string;
    count: number;
    icon: React.ReactNode;
    color: string;
}

const APPROVAL_CATEGORIES: ApprovalCategory[] = [
    { id: 'po', name: 'PURCHASE\nORDERS', count: 55, icon: <MdFolder />, color: '#10B981' },
    { id: 'ca', name: 'CREDIT\nAPPROVAL', count: 39, icon: <MdDescription />, color: '#F97316' },
    { id: 'se', name: 'SERVICE ENTRY', count: 23, icon: <MdShoppingCart />, color: '#06B6D4' },
    { id: 'jcs', name: 'JCS', count: 4, icon: <FaWrench />, color: '#9CA3AF' },
    { id: 'contracts', name: 'CONTRACTS', count: 2, icon: <MdReceiptLong />, color: '#FBBF24' },
    { id: 'partnerhub', name: 'PARTNERHUB', count: 2, icon: <MdDescription />, color: '#EF4444' },
    { id: 'shopping', name: 'SHOPPING CART', count: 0, icon: <MdShoppingCart />, color: '#EC4899' },
    { id: 're', name: 'REVENUE\nEXPENSES', count: 0, icon: <MdAttachMoney />, color: '#DC2626' },
    { id: 'rgp', name: 'RGP', count: 0, icon: <MdSwapHoriz />, color: '#F59E0B' },
    { id: 'nrgp', name: 'NRGP', count: 0, icon: <MdSwapHoriz />, color: '#EA580C' },
    { id: 'leaves', name: 'LEAVES', count: 0, icon: <MdReceiptLong />, color: '#DC2626' },
    { id: 'er', name: 'EMPLOYEE\nREIMBURSEMENT', count: 0, icon: <MdAirlineSeatReclineNormal />, color: '#D1663D' },
];

export default function ApprovalCategories() {
    const router = useRouter();

    const [showMenuPanel, setShowMenuPanel] = useState(false);
    const [showLogoutModal, setShowLogoutModal] = useState(false);

    const handleCategoryClick = (categoryId: string) => {
        router.push(`/mobile/approvals?category=${categoryId}`);
    };

    const totalPending = APPROVAL_CATEGORIES.reduce(
        (sum, cat) => sum + cat.count,
        0
    );

    const handleReportIssue = () => {
        setShowMenuPanel(false);
        router.push('/mobile/report-issue');
    };

    const handleLogoutConfirm = () => {
        setShowLogoutModal(false);
        // router.push('/login');
    };

    return (
        <div className={styles.container}>
            <div className={styles.header}>
                <span
                    className={styles.menuIcon}
                    onClick={() => setShowMenuPanel(true)}
                >
                    ☰
                </span>
                <h1 className={styles.headerTitle}>Approvals</h1>
                <span
                    className={styles.powerIcon}
                    onClick={() => setShowLogoutModal(true)}
                >
                    ⏻
                </span>
            </div>
            <div className={styles.categoriesGrid}>
                {APPROVAL_CATEGORIES.map((category) => (
                    <button
                        key={category.id}
                        className={styles.categoryRow}
                        onClick={() => handleCategoryClick(category.id)}
                    >
                        <div
                            className={styles.categoryIconWrapper}
                            style={{ color: category.color }}
                        >
                            {category.icon}
                        </div>

                        <div className={styles.categoryInfo}>
                            <span className={styles.categoryName}>{category.name}</span>
                        </div>

                        <div className={styles.rightWrap}>
                            {category.count > 0 && (
                                <span className={styles.categoryBadge}>{category.count}</span>
                            )}
                            <FiChevronRight className={styles.categoryArrow} />
                        </div>
                    </button>
                ))}
            </div>
            {showMenuPanel && (
                <div
                    className={styles.menuOverlay}
                    onClick={() => setShowMenuPanel(false)}
                >
                    <div
                        className={styles.menuPanel}
                        onClick={(e) => e.stopPropagation()}
                    >
                        <div className={styles.menuTopBar}>
                            <span className={styles.menuGreeting}>
                                Hello Md Nasrullah Khan !
                            </span>
                        </div>

                        <div className={styles.menuList}>
                            <button
                                className={styles.menuItem}
                                onClick={() => setShowMenuPanel(false)}
                            >
                                <div className={styles.menuItemIcon}>
                                    <MdDescription />
                                </div>
                                <span className={styles.menuItemTitle}>PENDING APPROVALS</span>
                                <span className={styles.menuItemBadge}>{totalPending}</span>
                            </button>

                            <button
                                className={styles.menuItem}
                                onClick={handleReportIssue}
                            >
                                <div className={styles.menuItemIcon}>
                                    <MdDescription />
                                </div>
                                <span className={styles.menuItemTitle}>REPORT ISSUE</span>
                            </button>
                        </div>

                        <div className={styles.menuVersion}>
                            Version 3.2.0.9.6 (QA)
                        </div>
                    </div>
                </div>
            )}
            {showLogoutModal && (
                <div
                    className={styles.modalOverlay}
                    onClick={() => setShowLogoutModal(false)}
                >
                    <div
                        className={styles.modalBox}
                        onClick={(e) => e.stopPropagation()}
                    >
                        <p className={styles.modalMessage}>
                            Are you sure you want to Logout?
                        </p>

                        <div className={styles.modalButtons}>
                            <button
                                className={styles.yesBtn}
                                onClick={handleLogoutConfirm}
                            >
                                Yes
                            </button>
                            <button
                                className={styles.noBtn}
                                onClick={() => setShowLogoutModal(false)}
                            >
                                No
                            </button>
                        </div>
                    </div>
                </div>
            )}
        </div>
    );
}
